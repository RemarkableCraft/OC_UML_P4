/* liste classe */
class Menu {
 - plat_menu : id_plat
 - qte_plat_menu : number
}

class Media {
 - nom_fichier : string
 - type_fichier : string
 - plat : id_plat
}

class Plat {
 - nom_plat : string
 - ingrédient_plat : array
 - prix_plat : number
 - type_plat : string
}

class Livreur {
 - statut_livreur : string
 - pseudo_livreur : string
 - mail_livreur : string
 - latitude : number
 - lontitude : number
 - téléphone : number
}

class Stock {
 - plat_stock : id_plat
 - livreur_stock : id_livreur
 - qte_plat_stock : number
}

class Client {
 - nom : string
 - prénom : string
 - pseudo_client : string
 - domicile_client : id_adresse
 - bureau_client : id_adresse
 - telephone_client : number
 - mail_client : string
}

class Adresse {
 - num_voie : number
 - nom_voie : string
 - code_postal : string
 - nom_ville : string
}

class Plat_commande {
 - id_plat
 - id_commande
 - qt_plat
}

class Commande {
 - client_cde : id_client
 - prix_cde : number
 - livreur_cde : array
 - tps_livraison_cde : time
 - etat_cde : string
}
